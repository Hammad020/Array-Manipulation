//DECLARATION OF SD ARRAY
DECLARE myARR : ARRAY [1:5] OF INTEGER

DIM i, X : INTEGER
 
//INITIALISATION OF SD ARRAY
FOR i = 1 to 5
    myARR[i] = 0
NEXT i

//POPULATION OF SD ARRAY
FOR i = 1 to 5
    INPUT "INPUT A NUMBER TO INSERT INTO ARRAY: ", X
    myARR[i] = X
END FOR

//OUTPUT AN ARRAY
FOR i = 1 to 5
    OUTPUT myARR[i]
END FOR
